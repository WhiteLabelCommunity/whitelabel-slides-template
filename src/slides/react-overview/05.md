# Sintassi JSX

```javascript
import React, { Component } from 'react';
import './App.css';

const helloWorld = 'Hello World';

class App extends Component {
  render() {
    return (
    <div className="App">
       <h2>{helloWorld}</h2>
    </div>);
  } 
}

export default App;
```
